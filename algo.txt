#include <stdio.h>
 
void bubbleSort(int arr[]){
    for(int i = 0; i < 3; i++){
        for(int j = i +1; j <3; j++){
            int aux;
            if(arr[i]>arr[j]){
                aux = arr[j];
                arr[j] = arr[i];
                arr[i] = aux;
            }
        }
    }
    return arr;
}

}
